class mpl_toolkits.axisartist.AxisArtistHelper¶

AxisArtistHelper should define following method with given APIs. Note that the first axes argument will be axes attribute of the caller artist.:

# LINE (spinal line?)

def get_line(self, axes):
    # path : Path
    return path

def get_line_transform(self, axes):
    # ...
    # trans : transform
    return trans

# LABEL

def get_label_pos(self, axes):
    # x, y : position
    return (x, y), trans

def get_label_offset_transform(self,                 axes,
        pad_points, fontprops, renderer,
        bboxes,
        ):
    # va : vertical alignment
    # ha : horizontal alignment
    # a : angle
    return trans, va, ha, a
